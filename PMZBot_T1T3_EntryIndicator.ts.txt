declare upper;
declare once_per_bar;

# Hidden anchor ties study scaling to the symbol price axis (prevents drift in a separate panel).
plot ScaleAnchor = close;
ScaleAnchor.Hide();

input Premarket = {default "PRE", "ALL"};
#hint futures: Use premarket or All when calculating PMZ.
input futures = no;
#hint futures: Plot PMZ after hours on futures.
input Banner = no;
#hint Banner: Debug labels (PMZ state, distance, gate). Off by default; distance-chart mode is visual.
input ShowStatusStrip = yes;
#hint ShowStatusStrip: Large high-contrast PMZ and distance-gate values in the chart corner.
input ShowDistanceChartStyle = yes;
#hint ShowDistanceChartStyle: Match the /distance chart: PMZ band, +/- distance gate, admit slivers, triangles.
input ShowEntryMarkers = yes;
#hint ShowEntryMarkers: Green/red triangles at the 1-minute close on entry fires. Requires a 1-minute /ES chart.
input ShowNeutralLines = yes;
#hint ShowNeutralLines: Dotted vertical lines when neutral is asserted (09:30 open, 5-min inside PMZ).
input AdjustPMZ = yes;
#hint AdjustPMZ: Adjust PMZ risk to be between 5 and 8 points. Production default is yes.
input ShowPMZBand = yes;
#hint ShowPMZBand: Plot the adjusted PMZ band. On by default; also enabled by ShowDistanceChartStyle.
input ShowOriginalPMZWhenAdjusted = no;
#hint ShowOriginalPMZWhenAdjusted: Plot the unadjusted PMZ band in gray when AdjustPMZ is enabled.
input MaxDistanceOutsidePmzPoints = 10.0;
#hint MaxDistanceOutsidePmzPoints: CALL distance is close - PMZ Low; PUT distance is PMZ High - close. Strict greater-than rejection.
input MaxTriggerCandles = 5;
#hint MaxTriggerCandles: One-minute trigger budget after a 5-minute inside-PMZ pullback confirmation.
input EnableFullReversalFreshBreakout = yes;
#hint EnableFullReversalFreshBreakout: Treat a 5-minute outside-to-opposite-outside traversal as a fresh clean breakout arm.
input EntryGateTimeET = 0935;
#hint EntryGateTimeET: Earliest ET candle close that may plot an entry marker.

def period = if GetAggregationPeriod() <= AggregationPeriod.FIVE_MIN then
    AggregationPeriod.FIVE_MIN
   else GetAggregationPeriod();

def tclose = 1600;
def topen = 0930;
def fut_close = 1800;
def nt2 = 2355;
def nt3 = 0000;
def nt4 = 0930;

def price = close(period = period);
def h = high(period = period);
def l = low(period = period);
def c = close(period = period);
def o = open(period = period);

def cnull = if period <= AggregationPeriod.FIVE_MIN then 1555
    else if period == AggregationPeriod.TEN_MIN then 1550
    else if period == AggregationPeriod.FIFTEEN_MIN then 1545
    else if period == AggregationPeriod.THIRTY_MIN then 1530
    else if period == AggregationPeriod.HOUR then 1500
    else 0;

def pnull = if period <= AggregationPeriod.FIVE_MIN then 0725
    else if period == AggregationPeriod.TEN_MIN then 0720
    else if period == AggregationPeriod.FIFTEEN_MIN then 0715
    else 0700;

plot color = Double.NaN;
color.Hide();
color.DefineColor("pbull", Color.CYAN);
color.DefineColor("pbear", Color.RED);
color.DefineColor("fbull", Color.GREEN);
color.DefineColor("fbear", Color.YELLOW);
color.DefineColor("original_gray", Color.GRAY);

def LIS = If(SecondsTillTime(tclose) == 0, price, LIS[1]);
def otime = If(SecondsTillTime(topen) == 0, o, otime[1]);
def gapAmount = if !IsNaN(otime) and !IsNaN(LIS) then otime - LIS else Double.NaN;
def nperiod;
def nperiod1 = SecondsFromTime(nt2) <= 0 and SecondsFromTime(cnull) > 0;
def nperiod2 = SecondsFromTime(nt3) >= 0 and SecondsFromTime(nt4) < 0;
def fut_time = SecondsFromTime(cnull) > 0 and SecondsFromTime(fut_close) < 0;

switch (Premarket) {
case "ALL":
    nperiod = nperiod1 or nperiod2;
case "PRE":
    nperiod = SecondsFromTime(pnull) > 0 and SecondsFromTime(nt4) < 0;
}

def pmh = if SecondsTillTime(cnull) == 0 then 0
   else if nperiod then
       if pmh[1] > h then pmh[1] else h
   else pmh[1];

def pmhTimeSec = if SecondsTillTime(cnull) == 0 then Double.NaN
    else if nperiod then
        if pmh[1] > h then pmhTimeSec[1] else SecondsFromTime(0)
    else pmhTimeSec[1];

def pml = if SecondsTillTime(cnull) == 0 then 100000
    else if nperiod then
        if pml[1] < l then pml[1] else l
    else pml[1];

def pmlTimeSec = if SecondsTillTime(cnull) == 0 then Double.NaN
    else if nperiod and l > 0 then
        if pml[1] < l then pmlTimeSec[1] else SecondsFromTime(0)
    else pmlTimeSec[1];

def gap = if nperiod then
    if o < LIS then 0 else 1
    else gap[1];

def pmzHighOriginal = if SecondsFromTime(pnull) > 0 and SecondsFromTime(cnull) < 0 then
    if gap then
        pmh - ((pmh - pml) * .2)
    else
        pml + ((pmh - pml) * .4)
else Double.NaN;

def pmzLowOriginal = if SecondsFromTime(pnull) > 0 and SecondsFromTime(cnull) < 0 then
    if gap then
        pmh - ((pmh - pml) * .4)
    else
        pml + ((pmh - pml) * .2)
else Double.NaN;

def upper = if SecondsFromTime(0930) >= 0 and SecondsFromTime(cnull) < 0 then
    if gap then
        pmh - ((pmh - pml) * .2)
    else
      pml + ((pmh - pml) * .2)
    else Double.NaN;

def lower = if SecondsFromTime(0930) >= 0 and SecondsFromTime(cnull) < 0 then
    if gap then
        pmh - ((pmh - pml) * .4) else pml + ((pmh - pml) * .4)
    else Double.NaN;

def original_pmz_risk = if !IsNaN(upper) and !IsNaN(lower) then AbsValue(upper - lower) else Double.NaN;
def adjusted_risk = if AdjustPMZ and !IsNaN(original_pmz_risk) then
    if original_pmz_risk < 5 then 5
    else if original_pmz_risk > 8 then 8
    else original_pmz_risk
    else original_pmz_risk;

def pmz_center = if !IsNaN(upper) and !IsNaN(lower) then (upper + lower) / 2 else Double.NaN;
def adjusted_upper = if AdjustPMZ and !IsNaN(pmz_center) and !IsNaN(adjusted_risk) then
    pmz_center + (adjusted_risk / 2)
    else upper;
def adjusted_lower = if AdjustPMZ and !IsNaN(pmz_center) and !IsNaN(adjusted_risk) then
    pmz_center - (adjusted_risk / 2)
    else lower;

def pmzHigh = if AdjustPMZ then adjusted_upper else pmzHighOriginal;
def pmzLow = if AdjustPMZ then adjusted_lower else pmzLowOriginal;
def pmzHighDisplay = pmzHigh;
def pmzLowDisplay = pmzLow;

def inRthSession = SecondsFromTime(0930) >= 0 and SecondsFromTime(cnull) < 0;
def pmzBandActive = (ShowPMZBand or ShowDistanceChartStyle) and inRthSession;
def distanceGateActive = ShowDistanceChartStyle and inRthSession;
def bandLocked = pmzBandActive and !IsNaN(pmzHigh) and !IsNaN(pmzLow);
def callGateCeiling = if bandLocked then pmzLow + MaxDistanceOutsidePmzPoints else Double.NaN;
def putGateFloor = if bandLocked then pmzHigh - MaxDistanceOutsidePmzPoints else Double.NaN;
def callSliverValid = distanceGateActive and callGateCeiling > pmzHigh;
def putSliverValid = distanceGateActive and putGateFloor < pmzLow;

plot OriginalPmzHigh = if ShowOriginalPMZWhenAdjusted and AdjustPMZ and bandLocked then upper else Double.NaN;
plot OriginalPmzLow = if ShowOriginalPMZWhenAdjusted and AdjustPMZ and bandLocked then lower else Double.NaN;
plot PmzBandHigh = if bandLocked then pmzHigh else Double.NaN;
plot PmzBandLow = if bandLocked then pmzLow else Double.NaN;
plot DistanceCallGate = if distanceGateActive and !IsNaN(callGateCeiling) then callGateCeiling else Double.NaN;
plot DistancePutGate = if distanceGateActive and !IsNaN(putGateFloor) then putGateFloor else Double.NaN;
plot CallSliverHigh = if callSliverValid then callGateCeiling else Double.NaN;
plot CallSliverLow = if callSliverValid then pmzHigh else Double.NaN;
plot PutSliverHigh = if putSliverValid then pmzLow else Double.NaN;
plot PutSliverLow = if putSliverValid then putGateFloor else Double.NaN;

def ahh = if SecondsTillTime(cnull) == 0 then 0
    else if fut_time then
        if ahh[1] > h then ahh[1] else h
    else ahh[1];

def ahl = if SecondsTillTime(cnull) == 0 then 100000
    else if fut_time then
        if ahl[1] < l then ahl[1] else l
    else ahl[1];

def ah_set = if fut_time then
    if o < LIS then 0 else 1
    else ah_set[1];

plot AfterHoursHigh = if ShowPMZBand and futures and (nperiod1 or nperiod2) then
    if ah_set then
        ahh - ((ahh - ahl) * .2) else ahl + ((ahh - ahl) * .2)
    else Double.NaN;

plot AfterHoursLow = if ShowPMZBand and futures and (nperiod1 or nperiod2) then
    if ah_set then ahh - ((ahh - ahl) * .4) else ahl + ((ahh - ahl) * .4)
    else Double.NaN;

PmzBandHigh.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
PmzBandLow.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
PmzBandHigh.SetDefaultColor(color.Color("pbull"));
PmzBandLow.SetDefaultColor(Color.MAGENTA);
PmzBandHigh.SetLineWeight(3);
PmzBandLow.SetLineWeight(3);

DistanceCallGate.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
DistancePutGate.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
DistanceCallGate.SetDefaultColor(Color.YELLOW);
DistancePutGate.SetDefaultColor(Color.ORANGE);
DistanceCallGate.SetStyle(Curve.LONG_DASH);
DistancePutGate.SetStyle(Curve.LONG_DASH);
DistanceCallGate.SetLineWeight(2);
DistancePutGate.SetLineWeight(2);

OriginalPmzHigh.SetDefaultColor(color.Color("original_gray"));
OriginalPmzLow.SetDefaultColor(color.Color("original_gray"));
OriginalPmzHigh.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
OriginalPmzLow.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
OriginalPmzHigh.Hide();
OriginalPmzLow.Hide();
AfterHoursHigh.SetDefaultColor(color.Color("fbull"));
AfterHoursLow.SetDefaultColor(color.Color("fbear"));
AfterHoursHigh.Hide();
AfterHoursLow.Hide();

AddCloud(PmzBandHigh, PmzBandLow, color.Color("pbull"), Color.MAGENTA, no);
AddCloud(CallSliverLow, CallSliverHigh, CreateColor(52, 168, 83), CreateColor(52, 168, 83), no);
AddCloud(PutSliverLow, PutSliverHigh, CreateColor(52, 168, 83), CreateColor(52, 168, 83), no);

script PmzEntryMachine {
    input prevCode = 0;
    input isRthOpen = 0;
    input isFiveMinuteClose = 0;
    input fiveSlot = 0;
    input entriesAllowed = 0;
    input oneAbove = 0;
    input oneBelow = 0;
    input fiveAbove = 0;
    input fiveBelow = 0;
    input fiveInside = 0;
    input fiveOverlapsPmz = 0;
    input callWithinDistance = 0;
    input putWithinDistance = 0;
    input fiveCallWithinDistance = 0;
    input fivePutWithinDistance = 0;
    input maxTriggerCandles = 5;
    input enableFullReversalFreshBreakout = 1;

    def rthOpen = isRthOpen != 0;
    def fiveCloseBar = isFiveMinuteClose != 0;
    def entriesOk = entriesAllowed != 0;
    def oneAboveBar = oneAbove != 0;
    def oneBelowBar = oneBelow != 0;
    def fiveAboveBar = fiveAbove != 0;
    def fiveBelowBar = fiveBelow != 0;
    def fiveInsideBar = fiveInside != 0;
    def fiveOverlapsBand = fiveOverlapsPmz != 0;
    def callDistanceOk = callWithinDistance != 0;
    def putDistanceOk = putWithinDistance != 0;
    def fiveCallDistanceOk = fiveCallWithinDistance != 0;
    def fivePutDistanceOk = fivePutWithinDistance != 0;
    def fullReversalEnabled = enableFullReversalFreshBreakout != 0;

    def statePrev = Floor(prevCode % 10);
    def armPrev = Floor(Floor(prevCode / 10) % 10);
    def budgetPrev = Floor(prevCode / 100);

    def crossedThroughOutside = fiveCloseBar and fiveOverlapsBand and (fiveAboveBar or fiveBelowBar);
    def stateForFiveMinute = if crossedThroughOutside then 0 else statePrev;
    def fullReversalToCall = fullReversalEnabled and stateForFiveMinute == 4 and fiveAboveBar;
    def fullReversalToPut = fullReversalEnabled and stateForFiveMinute == 3 and fiveBelowBar;
    def isFirstEntryGateBoundary = fiveCloseBar and fiveSlot == 1;
    def crossThroughCleanCall = fiveCloseBar and stateForFiveMinute == 2 and fiveAboveBar and fiveCallDistanceOk;
    def crossThroughCleanPut = fiveCloseBar and stateForFiveMinute == 1 and fiveBelowBar and fivePutDistanceOk;
    def sameBarCleanCall = fiveCloseBar and fiveAboveBar and fiveCallDistanceOk and
        (fullReversalToCall or (stateForFiveMinute == 0 and isFirstEntryGateBoundary) or crossThroughCleanCall);
    def sameBarCleanPut = fiveCloseBar and fiveBelowBar and fivePutDistanceOk and
        (fullReversalToPut or (stateForFiveMinute == 0 and isFirstEntryGateBoundary) or crossThroughCleanPut);

    def cleanCallFire = entriesOk and oneAboveBar and callDistanceOk and (armPrev == 1 or sameBarCleanCall);
    def cleanPutFire = entriesOk and oneBelowBar and putDistanceOk and (armPrev == 2 or sameBarCleanPut);
    def pullbackCallFire = entriesOk and statePrev == 1 and oneAboveBar and callDistanceOk;
    def pullbackPutFire = entriesOk and statePrev == 2 and oneBelowBar and putDistanceOk;

    def cleanArmConsumed = cleanCallFire or cleanPutFire or
        (armPrev == 1 and (oneBelowBar or (entriesOk and oneAboveBar and !callDistanceOk))) or
        (armPrev == 2 and (oneAboveBar or (entriesOk and oneBelowBar and !putDistanceOk)));

    def pullbackMiss = (statePrev == 1 or statePrev == 2) and !pullbackCallFire and !pullbackPutFire;
    def budgetAfterOneMinute = if pullbackMiss and !fiveCloseBar then budgetPrev - 1 else budgetPrev;
    def pullbackTimedOut = pullbackMiss and !fiveCloseBar and budgetAfterOneMinute <= 0;
    def armsCleanCall = rthOpen and oneAboveBar or
        (fiveCloseBar and fiveAboveBar and (stateForFiveMinute == 0 or fullReversalToCall));
    def armsCleanPut = rthOpen and oneBelowBar or
        (fiveCloseBar and fiveBelowBar and (stateForFiveMinute == 0 or fullReversalToPut));

    def nextState =
        if rthOpen then
            if oneAboveBar then 3 else if oneBelowBar then 4 else 0
        else if pullbackCallFire or cleanCallFire then 3
        else if pullbackPutFire or cleanPutFire then 4
        else if pullbackTimedOut then 0
        else if fiveCloseBar then
            if fiveInsideBar then
                if stateForFiveMinute == 3 then 1
                else if stateForFiveMinute == 4 then 2
                else stateForFiveMinute
            else if fiveAboveBar then 3
            else if fiveBelowBar then 4
            else stateForFiveMinute
        else statePrev;

    def nextArm =
        if cleanCallFire or cleanPutFire then 0
        else if armsCleanCall then 1
        else if armsCleanPut then 2
        else if cleanArmConsumed then 0
        else armPrev;

    def nextBudget =
        if rthOpen then 0
        else if pullbackCallFire or pullbackPutFire or cleanCallFire or cleanPutFire then 0
        else if pullbackTimedOut then 0
        else if fiveCloseBar and fiveInsideBar and stateForFiveMinute == 3 then maxTriggerCandles
        else if fiveCloseBar and fiveInsideBar and stateForFiveMinute == 4 then maxTriggerCandles
        else if pullbackMiss and !fiveCloseBar then budgetAfterOneMinute
        else budgetPrev;

    plot newCode = nextState + nextArm * 10 + nextBudget * 100;
}

# Entry state machine.
# State codes: 0 InPMZ_NoPullback, 1 InPMZ_CallPullback, 2 InPMZ_PutPullback,
# 3 OutsidePmz(Call), 4 OutsidePmz(Put). Clean-arm codes: 0 none, 1 Call, 2 Put.
def secFromOpen = SecondsFromTime(0930);
def fiveSlot = if secFromOpen >= 0 then Floor(secFromOpen / 300) else -1;
def isFiveMinuteClose = GetAggregationPeriod() <= AggregationPeriod.MIN and fiveSlot >= 1 and fiveSlot != fiveSlot[1];
def isRthOpen = SecondsTillTime(0930) == 0;
def entriesAllowed = SecondsFromTime(EntryGateTimeET) >= 0 and SecondsFromTime(cnull) < 0;

def oneClose = close(period = AggregationPeriod.MIN);
def fiveClose = close(period = AggregationPeriod.FIVE_MIN);
def fiveHigh = high(period = AggregationPeriod.FIVE_MIN);
def fiveLow = low(period = AggregationPeriod.FIVE_MIN);
def oneAbove = !IsNaN(pmzHigh) and oneClose > pmzHigh;
def oneBelow = !IsNaN(pmzLow) and oneClose < pmzLow;
def fiveAbove = !IsNaN(pmzHigh) and fiveClose > pmzHigh;
def fiveBelow = !IsNaN(pmzLow) and fiveClose < pmzLow;
def fiveInside = !IsNaN(pmzHigh) and !IsNaN(pmzLow) and fiveClose <= pmzHigh and fiveClose >= pmzLow;
def fiveOverlapsPmz = !IsNaN(pmzHigh) and !IsNaN(pmzLow) and fiveLow <= pmzHigh and fiveHigh >= pmzLow;

def callDistance = oneClose - pmzLow;
def putDistance = pmzHigh - oneClose;
def callWithinDistance = callDistance <= MaxDistanceOutsidePmzPoints;
def putWithinDistance = putDistance <= MaxDistanceOutsidePmzPoints;
def fiveCallDistance = fiveClose - pmzLow;
def fivePutDistance = pmzHigh - fiveClose;
def fiveCallWithinDistance = fiveCallDistance <= MaxDistanceOutsidePmzPoints;
def fivePutWithinDistance = fivePutDistance <= MaxDistanceOutsidePmzPoints;

def isRthOpenFlag = if isRthOpen then 1 else 0;
def isFiveMinuteCloseFlag = if isFiveMinuteClose then 1 else 0;
def entriesAllowedFlag = if entriesAllowed then 1 else 0;
def oneAboveFlag = if oneAbove then 1 else 0;
def oneBelowFlag = if oneBelow then 1 else 0;
def fiveAboveFlag = if fiveAbove then 1 else 0;
def fiveBelowFlag = if fiveBelow then 1 else 0;
def fiveInsideFlag = if fiveInside then 1 else 0;
def fiveOverlapsPmzFlag = if fiveOverlapsPmz then 1 else 0;
def callWithinDistanceFlag = if callWithinDistance then 1 else 0;
def putWithinDistanceFlag = if putWithinDistance then 1 else 0;
def fiveCallWithinDistanceFlag = if fiveCallWithinDistance then 1 else 0;
def fivePutWithinDistanceFlag = if fivePutWithinDistance then 1 else 0;
def enableFullReversalFlag = if EnableFullReversalFreshBreakout then 1 else 0;

def smState = CompoundValue(1,
    PmzEntryMachine(
        smState[1],
        isRthOpenFlag,
        isFiveMinuteCloseFlag,
        fiveSlot,
        entriesAllowedFlag,
        oneAboveFlag,
        oneBelowFlag,
        fiveAboveFlag,
        fiveBelowFlag,
        fiveInsideFlag,
        fiveOverlapsPmzFlag,
        callWithinDistanceFlag,
        putWithinDistanceFlag,
        fiveCallWithinDistanceFlag,
        fivePutWithinDistanceFlag,
        MaxTriggerCandles,
        enableFullReversalFlag).newCode,
    0);

def state = Floor(smState % 10);
def cleanArm = Floor(Floor(smState / 10) % 10);
def triggerBudget = Floor(smState / 100);

# Entry fires are computed at study scope so markers use native booleans, not script yes/no inputs.
def statePrev = Floor(smState[1] % 10);
def armPrev = Floor(Floor(smState[1] / 10) % 10);
def crossedThroughOutsideFire = isFiveMinuteClose and fiveOverlapsPmz and (fiveAbove or fiveBelow);
def stateForFiveMinuteFire = if crossedThroughOutsideFire then 0 else statePrev;
def fullReversalToCallFire = EnableFullReversalFreshBreakout and stateForFiveMinuteFire == 4 and fiveAbove;
def fullReversalToPutFire = EnableFullReversalFreshBreakout and stateForFiveMinuteFire == 3 and fiveBelow;
def isFirstEntryGateBoundaryFire = isFiveMinuteClose and fiveSlot == 1;
def crossThroughCleanCallFire = isFiveMinuteClose and stateForFiveMinuteFire == 2 and fiveAbove and fiveCallWithinDistance;
def crossThroughCleanPutFire = isFiveMinuteClose and stateForFiveMinuteFire == 1 and fiveBelow and fivePutWithinDistance;
def sameBarCleanCallFire = isFiveMinuteClose and fiveAbove and fiveCallWithinDistance and
    (fullReversalToCallFire or (stateForFiveMinuteFire == 0 and isFirstEntryGateBoundaryFire) or crossThroughCleanCallFire);
def sameBarCleanPutFire = isFiveMinuteClose and fiveBelow and fivePutWithinDistance and
    (fullReversalToPutFire or (stateForFiveMinuteFire == 0 and isFirstEntryGateBoundaryFire) or crossThroughCleanPutFire);
def cleanCallFire = entriesAllowed and oneAbove and callWithinDistance and (armPrev == 1 or sameBarCleanCallFire);
def cleanPutFire = entriesAllowed and oneBelow and putWithinDistance and (armPrev == 2 or sameBarCleanPutFire);
def pullbackCallFire = entriesAllowed and statePrev == 1 and oneAbove and callWithinDistance;
def pullbackPutFire = entriesAllowed and statePrev == 2 and oneBelow and putWithinDistance;
def callFire = cleanCallFire or pullbackCallFire;
def putFire = cleanPutFire or pullbackPutFire;
def pullbackFire = pullbackCallFire or pullbackPutFire;

# Entry markers: green/red triangles at the 1-minute close (same placement as /distance chart).
plot CallEntryMarker = if ShowEntryMarkers and callFire then oneClose else Double.NaN;
CallEntryMarker.SetPaintingStrategy(PaintingStrategy.ARROW_UP);
CallEntryMarker.SetLineWeight(5);
CallEntryMarker.SetDefaultColor(CreateColor(52, 168, 83));
CallEntryMarker.HideBubble();

plot PutEntryMarker = if ShowEntryMarkers and putFire then oneClose else Double.NaN;
PutEntryMarker.SetPaintingStrategy(PaintingStrategy.ARROW_DOWN);
PutEntryMarker.SetLineWeight(5);
PutEntryMarker.SetDefaultColor(CreateColor(234, 67, 53));
PutEntryMarker.HideBubble();

def entryTimeSec = SecondsFromTime(0);
AddChartBubble(
    ShowEntryMarkers and callFire,
    oneClose,
    "CALL " +
    AsText(
        Floor(entryTimeSec / 3600) + Floor((entryTimeSec % 3600) / 60) / 100,
        NumberFormat.TWO_DECIMAL_PLACES) + " ET",
    CreateColor(52, 168, 83),
    yes);
AddChartBubble(
    ShowEntryMarkers and putFire,
    oneClose,
    "PUT " +
    AsText(
        Floor(entryTimeSec / 3600) + Floor((entryTimeSec % 3600) / 60) / 100,
        NumberFormat.TWO_DECIMAL_PLACES) + " ET",
    CreateColor(234, 67, 53),
    no);

def neutralAtOpen = isRthOpen;
def neutralOnFiveInside = isFiveMinuteClose and fiveInside;
AddVerticalLine(ShowNeutralLines and neutralAtOpen, "Neutral", CreateColor(161, 66, 244), Curve.SHORT_DASH);
AddVerticalLine(ShowNeutralLines and neutralOnFiveInside, "Neutral", CreateColor(161, 66, 244), Curve.SHORT_DASH);

# Status strip — large font, bright colors (readable on TOS dark chart backgrounds).

    ShowStatusStrip,
    "PMZ Low: " + AsText(pmzLowDisplay, NumberFormat.TWO_DECIMAL_PLACES),
    Color.RED,
    location = Location.TOP_LEFT,

AddLabel(
    ShowStatusStrip,
    "PMZ High: " + AsText(pmzHighDisplay, NumberFormat.TWO_DECIMAL_PLACES),
    Color.CYAN,
    location = Location.TOP_LEFT,
    size = FontSize.LARGE);
AddLabel(
    ShowStatusStrip,
    "+/-" + AsText(MaxDistanceOutsidePmzPoints, NumberFormat.TWO_DECIMAL_PLACES) +
    " gate  Call " + AsText(callGateCeiling, NumberFormat.TWO_DECIMAL_PLACES) +
    "  Put " + AsText(putGateFloor, NumberFormat.TWO_DECIMAL_PLACES),
    Color.YELLOW,
    location = Location.TOP_LEFT,
    size = FontSize.LARGE);
AddLabel(
    ShowStatusStrip and (callFire or putFire),
    if callFire then "ENTRY CALL" else "ENTRY PUT",
    if callFire then Color.LIME else Color.ORANGE,
    location = Location.TOP_LEFT,
    size = FontSize.X_LARGE);
AddLabel(
    ShowStatusStrip,
    "1m " +
    (if oneAbove then "Above PMZ"
    else if oneBelow then "Below PMZ"
    else "Inside PMZ") +
    " | Arm " +
    (if cleanArm == 1 then "Call"
    else if cleanArm == 2 then "Put"
    else "None") +
    " | Dist " +
    AsText(callDistance, NumberFormat.TWO_DECIMAL_PLACES) + "/" +
    AsText(putDistance, NumberFormat.TWO_DECIMAL_PLACES),
    Color.LIME,
    location = Location.TOP_LEFT,
    size = FontSize.MEDIUM);

AddLabel(Banner,
    "LIS: " + AsText(LIS, NumberFormat.TWO_DECIMAL_PLACES) +
    " | RTH Open: " + AsText(otime, NumberFormat.TWO_DECIMAL_PLACES) +
    " | Gap: " + AsText(gapAmount, NumberFormat.TWO_DECIMAL_PLACES) +
    (if gap == 1 then " (UP)" else " (DOWN/FLAT)"),
    if gap == 1 then Color.GREEN else Color.RED);

AddLabel(Banner,
    "PMH Time: " +
    (if IsNaN(pmhTimeSec) then "n/a" else
        AsText(
