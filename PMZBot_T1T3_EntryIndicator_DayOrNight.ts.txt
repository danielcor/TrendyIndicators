declare upper;
declare once_per_bar;

# Hidden anchor ties study scaling to the symbol price axis (prevents drift in a separate panel).
plot ScaleAnchor = close;
ScaleAnchor.Hide();

input Premarket = {default "PRE", "ALL"};
#hint Premarket: Day premarket window for day PMZ (PRE = cash premarket; ALL = legacy overnight day accumulation).
input futures = no;
#hint futures: When NightMode is off, show the night/AH PMZ band without night entry markers (Tr3ndy-style).
input NightMode = no;
#hint NightMode: Night-only mode. Night band + night entries in the Globex night window; blank during cash RTH. Suppresses day band/entries.
input Banner = no;
#hint Banner: Debug labels (PMZ state, distance, gate). Off by default; distance-chart mode is visual.
input ShowStatusStrip = yes;
#hint ShowStatusStrip: Large high-contrast PMZ and distance-gate values in the chart corner.
input ShowDistanceChartStyle = yes;
#hint ShowDistanceChartStyle: Match the /distance chart: PMZ band, +/- distance gate, admit slivers, triangles.
input ShowEntryMarkers = yes;
#hint ShowEntryMarkers: Green/red triangles at the 1-minute close on entry fires. Requires a 1-minute /ES chart.
input ShowNeutralLines = yes;
#hint ShowNeutralLines: Dotted vertical lines when neutral is asserted (session open, 5-min inside PMZ).
input AdjustPMZ = yes;
#hint AdjustPMZ: Clamp PMZ width to [MinPmzWidthPoints, MaxPmzWidthPoints] (center-preserving). Production default yes.
input MinPmzWidthPoints = 5.0;
#hint MinPmzWidthPoints: Expand band if original risk is narrower. Day bot parity default 5.
input MaxPmzWidthPoints = 8.0;
#hint MaxPmzWidthPoints: Contract band if original risk is wider. Day bot parity default 8. If Min > Max, clamp is skipped.
input ShowPMZBand = yes;
#hint ShowPMZBand: Plot the adjusted PMZ band. On by default; also enabled by ShowDistanceChartStyle.
input ShowOriginalPMZWhenAdjusted = no;
#hint ShowOriginalPMZWhenAdjusted: Plot the unadjusted PMZ band in gray when AdjustPMZ is enabled.
input MaxDistanceOutsidePmzPoints = 10.0;
#hint MaxDistanceOutsidePmzPoints: CALL distance is close - PMZ Low; PUT distance is PMZ High - close. Strict greater-than rejection.
input MaxTriggerCandles = 5;
#hint MaxTriggerCandles: One-minute trigger budget after a 5-minute inside-PMZ pullback confirmation.
input EnableFullReversalFreshBreakout = yes;
#hint EnableFullReversalFreshBreakout: Treat a 5-minute outside-to-opposite-outside traversal as a fresh clean breakout arm.
input EntryGateTimeET = 0935;
#hint EntryGateTimeET: Day mode only — HHmm offset from cash RegularTradingStart (0935 = +5 minutes). Not platform clock.
input NightFormExtensionMinutes = 60;
#hint NightFormExtensionMinutes: If classic 15:55-18:00 form has no valid H/L (e.g. Sunday), form from Globex reopen for this many minutes (clamped 0-240).

def period = if GetAggregationPeriod() <= AggregationPeriod.FIVE_MIN then
    AggregationPeriod.FIVE_MIN
   else GetAggregationPeriod();

def price = close(period = period);
def h = high(period = period);
def l = low(period = period);
def o = open(period = period);

# -----------------------------------------------------------------------------
# Cash RTH anchors (absolute epoch ms). On /ES these are cash 09:30-16:00 style
# boundaries, NOT Globex 18:00-17:00. Night epochs are DERIVED as fixed offsets
# from these so the study is independent of platform display timezone.
# -----------------------------------------------------------------------------
def rthStartEpoch = RegularTradingStart(GetYYYYMMDD());
def rthEndEpoch = RegularTradingEnd(GetYYYYMMDD());
def nowEpoch = GetTime();
def minuteMs = 60 * 1000;
def hourMs = 60 * minuteMs;
def dayMs = 24 * hourMs;

def entryGateMinutesSinceMidnight = Floor(EntryGateTimeET / 100) * 60 + EntryGateTimeET % 100;
def dayEntryGateEpoch = if !IsNaN(rthStartEpoch) then
    rthStartEpoch + (entryGateMinutesSinceMidnight - (9 * 60 + 30)) * minuteMs
    else Double.NaN;
def premarketStartEpoch = if !IsNaN(rthStartEpoch) then rthStartEpoch - 125 * minuteMs else Double.NaN;
def premarketResetEpoch = if !IsNaN(rthEndEpoch) then rthEndEpoch - 5 * minuteMs else Double.NaN;
def lisResetEpoch = rthEndEpoch;

# Cash RTH window used for blank-when-NightMode and day band.
def inCashRth = !IsNaN(rthStartEpoch) and !IsNaN(premarketResetEpoch) and
    nowEpoch >= rthStartEpoch and nowEpoch < premarketResetEpoch;
def isCashOpenPulse = !IsNaN(rthStartEpoch) and nowEpoch >= rthStartEpoch and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < rthStartEpoch);

# Carry last known cash close for the active night session only.
# Latch at form start (cash close - 5m); clear at next cash open so mid-day does not keep Friday's base.
def lastRthEndEpoch = CompoundValue(1,
    if isCashOpenPulse then Double.NaN
    else if !IsNaN(rthEndEpoch) and nowEpoch >= (rthEndEpoch - 5 * minuteMs) then rthEndEpoch
    else if !IsNaN(lastRthEndEpoch[1]) then lastRthEndEpoch[1]
    else Double.NaN,
    Double.NaN);

# Midnight anchor for bubble clock labels. Weekend: advance last cash-day midnight to the bar's day.
def fridayMidnightEpoch = if !IsNaN(lastRthEndEpoch) then lastRthEndEpoch - 16 * hourMs else Double.NaN;
def midnightEtEpoch = if !IsNaN(rthStartEpoch) then rthStartEpoch - (9 * 60 + 30) * minuteMs
    else if !IsNaN(fridayMidnightEpoch) and !IsNaN(nowEpoch) then
        fridayMidnightEpoch + Floor((nowEpoch - fridayMidnightEpoch) / dayMs) * dayMs
    else Double.NaN;

# Night epoch table (docs: ET walls; runtime: offsets from lastRthEnd / next cash open).
# Mon-Thu reopen = cash close + 2h. Fri close -> Sun 18:00 ~= +50h.
# Weekday reopen only if the bar is NEAR that epoch (else Sunday falsely matches Fri 18:00).
def nightFormStartEpoch = if !IsNaN(lastRthEndEpoch) then lastRthEndEpoch - 5 * minuteMs else Double.NaN;
def weekdayReopenEpoch = if !IsNaN(lastRthEndEpoch) then lastRthEndEpoch + 2 * hourMs else Double.NaN;
# Fri 16:00 wall → Sun 18:00 wall: same calendar offset as two midnights + 18h from Fri midnight.
# Uses dayMs-normalized midnights so spring-forward does not skew absolute +50h.
def weekendReopenEpoch = if !IsNaN(fridayMidnightEpoch) then
    fridayMidnightEpoch + 2 * dayMs + 18 * hourMs
    else Double.NaN;
def reopenProximityMs = 3 * hourMs;

def isWeekdayReopen = !IsNaN(weekdayReopenEpoch) and
    nowEpoch >= weekdayReopenEpoch and nowEpoch < weekdayReopenEpoch + reopenProximityMs and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < weekdayReopenEpoch);
def isWeekendReopen = !IsNaN(weekendReopenEpoch) and
    nowEpoch >= weekendReopenEpoch and nowEpoch < weekendReopenEpoch + reopenProximityMs and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < weekendReopenEpoch);
# Long-gap: first bar after dark that lands within 12h after expected weekday/weekend reopen
# (covers late first prints outside the tight 3h pulse window, not mid-session holes).
# Shared next-cash-open close for night display/entries (weekend fallback when rthStart NaN).
def nightEntryCloseEpoch = if !IsNaN(rthStartEpoch) then
    (if nowEpoch < rthStartEpoch then rthStartEpoch else rthStartEpoch + dayMs)
    else if !IsNaN(lastRthEndEpoch) then lastRthEndEpoch + (65 * hourMs + 30 * minuteMs)
    else Double.NaN;
def longGapReopen = !IsNaN(nowEpoch[1]) and (nowEpoch - nowEpoch[1]) >= 12 * hourMs and
    !IsNaN(lastRthEndEpoch) and nowEpoch > lastRthEndEpoch and
    (IsNaN(nightEntryCloseEpoch) or nowEpoch < nightEntryCloseEpoch) and
    (( !IsNaN(weekdayReopenEpoch) and nowEpoch >= weekdayReopenEpoch and
        (nowEpoch - weekdayReopenEpoch) <= 12 * hourMs) or
     ( !IsNaN(weekendReopenEpoch) and nowEpoch >= weekendReopenEpoch and
        (nowEpoch - weekendReopenEpoch) <= 12 * hourMs));
def isGlobexReopen = isWeekdayReopen or isWeekendReopen or longGapReopen;

def nightFormReset = !IsNaN(nightFormStartEpoch) and nowEpoch >= nightFormStartEpoch and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < nightFormStartEpoch);

# Effective reopen epoch for gate/slots.
def globexReopenEpoch = CompoundValue(1,
    if isGlobexReopen then
        (if isWeekdayReopen then weekdayReopenEpoch
        else if isWeekendReopen then weekendReopenEpoch
        else nowEpoch)
    else if nightFormReset or isCashOpenPulse then Double.NaN
    else globexReopenEpoch[1],
    Double.NaN);

def nightEntryOpenEpoch = if !IsNaN(globexReopenEpoch) then globexReopenEpoch + 5 * minuteMs else Double.NaN;

def inPrimaryNightForm = !IsNaN(nightFormStartEpoch) and !IsNaN(weekdayReopenEpoch) and
    nowEpoch > nightFormStartEpoch and nowEpoch < weekdayReopenEpoch;

# Night display: primary form window, or after reopen until next cash open, but not all weekend
# after a Friday phantom reopen (cap post-reopen display at 16h unless still before cash open overnight).
def inNightDisplay = !inCashRth and (
    inPrimaryNightForm or
    (!IsNaN(globexReopenEpoch) and !IsNaN(nightEntryCloseEpoch) and
     nowEpoch >= globexReopenEpoch and nowEpoch < nightEntryCloseEpoch and
     (nowEpoch - globexReopenEpoch) <= 16 * hourMs));

plot color = Double.NaN;
color.Hide();
color.DefineColor("pbull", Color.CYAN);
color.DefineColor("pbear", Color.RED);
color.DefineColor("fbull", Color.GREEN);
color.DefineColor("fbear", Color.YELLOW);
color.DefineColor("original_gray", Color.GRAY);

def LIS = if !IsNaN(lisResetEpoch) and nowEpoch >= lisResetEpoch and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < lisResetEpoch) then price else LIS[1];
def otime = if !IsNaN(rthStartEpoch) and nowEpoch >= rthStartEpoch and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < rthStartEpoch) then o else otime[1];
def gapAmount = if !IsNaN(otime) and !IsNaN(LIS) then otime - LIS else Double.NaN;

# Day Premarket=ALL still uses platform-clock overnight (legacy); PRE uses epoch premarket.
def nperiod;
def nperiod1Legacy = SecondsFromTime(2355) <= 0 and SecondsFromTime(1555) > 0;
def nperiod2Legacy = SecondsFromTime(0000) >= 0 and SecondsFromTime(0930) < 0;
switch (Premarket) {
case "ALL":
    nperiod = nperiod1Legacy or nperiod2Legacy;
case "PRE":
    nperiod = !IsNaN(premarketStartEpoch) and !IsNaN(rthStartEpoch) and
        nowEpoch > premarketStartEpoch and nowEpoch < rthStartEpoch;
}

def pmzReset = !IsNaN(premarketResetEpoch) and nowEpoch >= premarketResetEpoch and
    (IsNaN(nowEpoch[1]) or nowEpoch[1] < premarketResetEpoch);

def pmh = if pmzReset then 0
   else if nperiod then
       if pmh[1] > h then pmh[1] else h
   else pmh[1];

def pmhTimeSec = if pmzReset then Double.NaN
    else if nperiod then
        if pmh[1] > h then pmhTimeSec[1]
        else if !IsNaN(midnightEtEpoch) then Floor((nowEpoch - midnightEtEpoch) / 1000)
        else Double.NaN
    else pmhTimeSec[1];

def pml = if pmzReset then 100000
    else if nperiod then
        if pml[1] < l then pml[1] else l
    else pml[1];

def pmlTimeSec = if pmzReset then Double.NaN
    else if nperiod and l > 0 then
        if pml[1] < l then pmlTimeSec[1]
        else if !IsNaN(midnightEtEpoch) then Floor((nowEpoch - midnightEtEpoch) / 1000)
        else Double.NaN
    else pmlTimeSec[1];

def gap = if nperiod then
    if o < LIS then 0 else 1
    else gap[1];

# Tr3ndy RTH legs (gap-down may invert upper/lower plot assignment).
# Active band always orders high >= low so inside/distance match bot semantics.
def dayUpper = if inCashRth then
    if gap then
        pmh - ((pmh - pml) * .2)
    else
      pml + ((pmh - pml) * .2)
    else Double.NaN;

def dayLower = if inCashRth then
    if gap then
        pmh - ((pmh - pml) * .4) else pml + ((pmh - pml) * .4)
    else Double.NaN;

def dayBandHighRaw = if !IsNaN(dayUpper) and !IsNaN(dayLower) then
    (if dayUpper >= dayLower then dayUpper else dayLower) else dayUpper;
def dayBandLowRaw = if !IsNaN(dayUpper) and !IsNaN(dayLower) then
    (if dayUpper >= dayLower then dayLower else dayUpper) else dayLower;

def clampInputsValid = MinPmzWidthPoints <= MaxPmzWidthPoints;
def clampEnabled = AdjustPMZ and clampInputsValid;

def dayOriginalRisk = if !IsNaN(dayBandHighRaw) and !IsNaN(dayBandLowRaw) then
    AbsValue(dayBandHighRaw - dayBandLowRaw) else Double.NaN;
def dayAdjustedRisk = if clampEnabled and !IsNaN(dayOriginalRisk) then
    if dayOriginalRisk < MinPmzWidthPoints then MinPmzWidthPoints
    else if dayOriginalRisk > MaxPmzWidthPoints then MaxPmzWidthPoints
    else dayOriginalRisk
    else dayOriginalRisk;

def dayPmzCenter = if !IsNaN(dayBandHighRaw) and !IsNaN(dayBandLowRaw) then
    (dayBandHighRaw + dayBandLowRaw) / 2 else Double.NaN;
def dayAdjustedUpper = if clampEnabled and !IsNaN(dayPmzCenter) and !IsNaN(dayAdjustedRisk) then
    dayPmzCenter + (dayAdjustedRisk / 2)
    else dayBandHighRaw;
def dayAdjustedLower = if clampEnabled and !IsNaN(dayPmzCenter) and !IsNaN(dayAdjustedRisk) then
    dayPmzCenter - (dayAdjustedRisk / 2)
    else dayBandLowRaw;

def dayPmzHigh = if clampEnabled then dayAdjustedUpper else dayBandHighRaw;
def dayPmzLow = if clampEnabled then dayAdjustedLower else dayBandLowRaw;

# -----------------------------------------------------------------------------
# Night form (Tr3ndy primary 15:55-18:00 via epochs) + Peakbot extension after reopen
# when primary H/L is invalid (typical Sunday Globex open).
# -----------------------------------------------------------------------------
def ahhPrimary = if nightFormReset then 0
    else if inPrimaryNightForm then
        if ahhPrimary[1] > h then ahhPrimary[1] else h
    else ahhPrimary[1];

def ahlPrimary = if nightFormReset then 100000
    else if inPrimaryNightForm then
        if ahlPrimary[1] < l then ahlPrimary[1] else l
    else ahlPrimary[1];

def ahSetPrimary = if inPrimaryNightForm then
    if o < LIS then 0 else 1
    else ahSetPrimary[1];

def primaryFormValid = ahhPrimary > 0 and ahlPrimary > 0 and ahlPrimary < 100000 and ahhPrimary > ahlPrimary;
# Primary form must be recent enough for this reopen (not Friday H/L reused on Sunday).
def primaryFormFresh = primaryFormValid and !IsNaN(nightFormStartEpoch) and
    nowEpoch - nightFormStartEpoch <= (6 * hourMs);
# Weekend reopen always extends; long-gap only when primary is not fresh (preserve weekday primary).
def forceExtensionReopen = isWeekendReopen or (longGapReopen and !primaryFormFresh);

# Lock at reopen: force-extension clears primary; else require fresh primary.
def primaryValidLocked = CompoundValue(1,
    if isGlobexReopen then
        (if forceExtensionReopen then 0 else if primaryFormFresh then 1 else 0)
    else if nightFormReset or isCashOpenPulse then 0
    else primaryValidLocked[1],
    0);

def nightFormExtensionMinutesClamped =
    if NightFormExtensionMinutes < 0 then 0
    else if NightFormExtensionMinutes > 240 then 240
    else NightFormExtensionMinutes;
def extensionEndEpoch = if !IsNaN(globexReopenEpoch) then
    globexReopenEpoch + nightFormExtensionMinutesClamped * minuteMs
    else Double.NaN;
def inExtensionForm = primaryValidLocked == 0 and !IsNaN(globexReopenEpoch) and !IsNaN(extensionEndEpoch) and
    nowEpoch >= globexReopenEpoch and nowEpoch < extensionEndEpoch;

def ahhExt = if isGlobexReopen and primaryValidLocked == 0 then h
    else if inExtensionForm then
        if ahhExt[1] > h then ahhExt[1] else h
    else ahhExt[1];

def ahlExt = if isGlobexReopen and primaryValidLocked == 0 then l
    else if inExtensionForm then
        if ahlExt[1] < l then ahlExt[1] else l
    else ahlExt[1];

def ahSetExt = if isGlobexReopen and primaryValidLocked == 0 then
    (if o < LIS then 0 else 1)
    else if inExtensionForm then
        (if o < LIS then 0 else 1)
    else ahSetExt[1];

def ahh = if primaryValidLocked == 1 then ahhPrimary
    else if inExtensionForm or (primaryValidLocked == 0 and !IsNaN(globexReopenEpoch) and nowEpoch >= globexReopenEpoch) then ahhExt
    else ahhPrimary;
def ahl = if primaryValidLocked == 1 then ahlPrimary
    else if inExtensionForm or (primaryValidLocked == 0 and !IsNaN(globexReopenEpoch) and nowEpoch >= globexReopenEpoch) then ahlExt
    else ahlPrimary;
def ah_set = if primaryValidLocked == 1 then ahSetPrimary
    else if inExtensionForm or (primaryValidLocked == 0 and !IsNaN(globexReopenEpoch) and nowEpoch >= globexReopenEpoch) then ahSetExt
    else ahSetPrimary;

def nightFormValid = ahh > 0 and ahl > 0 and ahl < 100000 and ahh > ahl;
# Extension path active after reopen with primary not locked (even if H/L still invalid).
def nightExtensionPath = primaryValidLocked == 0 and !IsNaN(globexReopenEpoch) and
    nowEpoch >= globexReopenEpoch;
# Freeze band before entries: primary locked, or extension window finished.
def nightFormFrozen = nightFormValid and (
    primaryValidLocked == 1 or
    (!IsNaN(extensionEndEpoch) and nowEpoch >= extensionEndEpoch));

def nightLegA = if nightFormValid then
    (if ah_set then ahh - ((ahh - ahl) * .2) else ahl + ((ahh - ahl) * .2))
    else Double.NaN;
def nightLegB = if nightFormValid then
    (if ah_set then ahh - ((ahh - ahl) * .4) else ahl + ((ahh - ahl) * .4))
    else Double.NaN;
def nightUpperRaw = if !IsNaN(nightLegA) and !IsNaN(nightLegB) then
    (if nightLegA >= nightLegB then nightLegA else nightLegB) else nightLegA;
def nightLowerRaw = if !IsNaN(nightLegA) and !IsNaN(nightLegB) then
    (if nightLegA >= nightLegB then nightLegB else nightLegA) else nightLegB;

def nightOriginalRisk = if !IsNaN(nightUpperRaw) and !IsNaN(nightLowerRaw) then
    AbsValue(nightUpperRaw - nightLowerRaw) else Double.NaN;
def nightAdjustedRisk = if clampEnabled and !IsNaN(nightOriginalRisk) then
    if nightOriginalRisk < MinPmzWidthPoints then MinPmzWidthPoints
    else if nightOriginalRisk > MaxPmzWidthPoints then MaxPmzWidthPoints
    else nightOriginalRisk
    else nightOriginalRisk;
def nightCenter = if !IsNaN(nightUpperRaw) and !IsNaN(nightLowerRaw) then
    (nightUpperRaw + nightLowerRaw) / 2 else Double.NaN;
def nightUpperAdj = if clampEnabled and !IsNaN(nightCenter) and !IsNaN(nightAdjustedRisk) then
    nightCenter + (nightAdjustedRisk / 2) else nightUpperRaw;
def nightLowerAdj = if clampEnabled and !IsNaN(nightCenter) and !IsNaN(nightAdjustedRisk) then
    nightCenter - (nightAdjustedRisk / 2) else nightLowerRaw;

def nightPmzHigh = if clampEnabled then nightUpperAdj else nightUpperRaw;
def nightPmzLow = if clampEnabled then nightLowerAdj else nightLowerRaw;

# Active band selection
# NightMode=yes: night band only inside night display; blank in cash RTH (no hold).
# NightMode=no: day band in cash RTH; night band when futures and inNightDisplay.
def showNightBand = (NightMode and inNightDisplay) or (!NightMode and futures and inNightDisplay);
def showDayBand = !NightMode and inCashRth;

def pmzHigh = if NightMode then
    (if showNightBand and nightFormValid then nightPmzHigh else Double.NaN)
    else if showDayBand then dayPmzHigh
    else if showNightBand and nightFormValid then nightPmzHigh
    else Double.NaN;
def pmzLow = if NightMode then
    (if showNightBand and nightFormValid then nightPmzLow else Double.NaN)
    else if showDayBand then dayPmzLow
    else if showNightBand and nightFormValid then nightPmzLow
    else Double.NaN;

def rawHighForOriginal = if showNightBand then nightUpperRaw else dayBandHighRaw;
def rawLowForOriginal = if showNightBand then nightLowerRaw else dayBandLowRaw;
def original_pmz_risk = if showNightBand then nightOriginalRisk else dayOriginalRisk;
def adjusted_risk = if showNightBand then nightAdjustedRisk else dayAdjustedRisk;

def pmzBandActive = (ShowPMZBand or ShowDistanceChartStyle) and
    ((showDayBand and !IsNaN(dayPmzHigh)) or (showNightBand and nightFormValid));
def distanceGateActive = ShowDistanceChartStyle and
    ((showDayBand and !IsNaN(dayPmzHigh)) or (showNightBand and nightFormValid));
def bandLocked = pmzBandActive and !IsNaN(pmzHigh) and !IsNaN(pmzLow);
def callGateCeiling = if bandLocked then pmzLow + MaxDistanceOutsidePmzPoints else Double.NaN;
def putGateFloor = if bandLocked then pmzHigh - MaxDistanceOutsidePmzPoints else Double.NaN;
def callSliverValid = distanceGateActive and callGateCeiling > pmzHigh;
def putSliverValid = distanceGateActive and putGateFloor < pmzLow;

plot OriginalPmzHigh = if ShowOriginalPMZWhenAdjusted and AdjustPMZ and bandLocked then rawHighForOriginal else Double.NaN;
plot OriginalPmzLow = if ShowOriginalPMZWhenAdjusted and AdjustPMZ and bandLocked then rawLowForOriginal else Double.NaN;
plot PmzBandHigh = if bandLocked then pmzHigh else Double.NaN;
plot PmzBandLow = if bandLocked then pmzLow else Double.NaN;
plot DistanceCallGate = if distanceGateActive and !IsNaN(callGateCeiling) then callGateCeiling else Double.NaN;
plot DistancePutGate = if distanceGateActive and !IsNaN(putGateFloor) then putGateFloor else Double.NaN;
plot CallSliverHigh = if callSliverValid then callGateCeiling else Double.NaN;
plot CallSliverLow = if callSliverValid then pmzHigh else Double.NaN;
plot PutSliverHigh = if putSliverValid then pmzLow else Double.NaN;
plot PutSliverLow = if putSliverValid then putGateFloor else Double.NaN;

PmzBandHigh.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
PmzBandLow.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
PmzBandHigh.SetDefaultColor(color.Color("pbull"));
PmzBandLow.SetDefaultColor(Color.MAGENTA);
PmzBandHigh.SetLineWeight(3);
PmzBandLow.SetLineWeight(3);

DistanceCallGate.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
DistancePutGate.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
DistanceCallGate.SetDefaultColor(Color.YELLOW);
DistancePutGate.SetDefaultColor(Color.ORANGE);
DistanceCallGate.SetStyle(Curve.LONG_DASH);
DistancePutGate.SetStyle(Curve.LONG_DASH);
DistanceCallGate.SetLineWeight(2);
DistancePutGate.SetLineWeight(2);

OriginalPmzHigh.SetDefaultColor(color.Color("original_gray"));
OriginalPmzLow.SetDefaultColor(color.Color("original_gray"));
OriginalPmzHigh.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);
OriginalPmzLow.SetPaintingStrategy(PaintingStrategy.HORIZONTAL);

# AddCloud colors must be compile-time constants (no dynamic if/showNightBand).
AddCloud(OriginalPmzHigh, OriginalPmzLow, color.Color("original_gray"), color.Color("original_gray"), yes);
AddCloud(PmzBandHigh, PmzBandLow, color.Color("pbull"), Color.MAGENTA, no);
AddCloud(CallSliverLow, CallSliverHigh, CreateColor(52, 168, 83), CreateColor(52, 168, 83), no);
AddCloud(PutSliverLow, PutSliverHigh, CreateColor(52, 168, 83), CreateColor(52, 168, 83), no);

script PmzEntryMachine {
    input prevCode = 0;
    input isSessionOpen = 0;
    input isFiveMinuteClose = 0;
    input isFirstFiveSlot = 0;
    input entriesAllowed = 0;
    input oneAbove = 0;
    input oneBelow = 0;
    input fiveAbove = 0;
    input fiveBelow = 0;
    input fiveInside = 0;
    input fiveOverlapsPmz = 0;
    input callWithinDistance = 0;
    input putWithinDistance = 0;
    input fiveCallWithinDistance = 0;
    input fivePutWithinDistance = 0;
    input maxTriggerCandles = 5;
    input enableFullReversalFreshBreakout = 1;
    input isLastMinuteOfFive = 0;

    def sessionOpen = isSessionOpen != 0;
    def fiveCloseBar = isFiveMinuteClose != 0;
    def lastMinuteOfFive = isLastMinuteOfFive != 0;
    def entriesOk = entriesAllowed != 0;
    def oneAboveBar = oneAbove != 0;
    def oneBelowBar = oneBelow != 0;
    def fiveAboveBar = fiveAbove != 0;
    def fiveBelowBar = fiveBelow != 0;
    def fiveInsideBar = fiveInside != 0;
    def fiveOverlapsBand = fiveOverlapsPmz != 0;
    def callDistanceOk = callWithinDistance != 0;
    def putDistanceOk = putWithinDistance != 0;
    def fiveCallDistanceOk = fiveCallWithinDistance != 0;
    def fivePutDistanceOk = fivePutWithinDistance != 0;
    def fullReversalEnabled = enableFullReversalFreshBreakout != 0;

    # Packed: state + 10*budget + 100*neutral + 1000*tookCall + 10000*tookPut
    # neutral = _lastFiveMinClosedInsidePmz; took* = hasHadAcceptedEntry* until next neutral.
    # T5 out of scope. Markers = PMZ SM fires (T1–T3 fan-out; T4 same signal).
    def statePrev = if IsNaN(prevCode) then 0 else Floor(prevCode % 10);
    def budgetPrev = if IsNaN(prevCode) then 0 else Floor(Floor(prevCode / 10) % 10);
    def neutralPrev = if IsNaN(prevCode) then 0 else Floor(Floor(prevCode / 100) % 10);
    def tookCallPrev = if IsNaN(prevCode) then 0 else Floor(Floor(prevCode / 1000) % 10);
    def tookPutPrev = if IsNaN(prevCode) then 0 else Floor(prevCode / 10000);

    def fullReversalToCall = fullReversalEnabled and statePrev == 4 and fiveAboveBar;
    def fullReversalToPut = fullReversalEnabled and statePrev == 3 and fiveBelowBar;
    def isFirstEntryGateBoundary = fiveCloseBar and isFirstFiveSlot != 0;
    def crossThroughCleanCall = fiveCloseBar and statePrev == 2 and fiveAboveBar and fiveCallDistanceOk;
    def crossThroughCleanPut = fiveCloseBar and statePrev == 1 and fiveBelowBar and fivePutDistanceOk;
    def sameBarCleanCall = fiveCloseBar and fiveAboveBar and fiveCallDistanceOk and
        (fullReversalToCall or (statePrev == 0 and isFirstEntryGateBoundary) or crossThroughCleanCall);
    def sameBarCleanPut = fiveCloseBar and fiveBelowBar and fivePutDistanceOk and
        (fullReversalToPut or (statePrev == 0 and isFirstEntryGateBoundary) or crossThroughCleanPut);

    # Type A: 1-min outside after neutral. Not on 5-min boundary bar; not on last 1-min of a
    # 5-min slot (bot processes that close with the five handler first — neutral may clear).
    # Same-side blocked until next neutral (took*).
    def typeAFromNeutralCall = neutralPrev == 1 and !fiveCloseBar and !lastMinuteOfFive and
        oneAboveBar and callDistanceOk and tookCallPrev == 0;
    def typeAFromNeutralPut = neutralPrev == 1 and !fiveCloseBar and !lastMinuteOfFive and
        oneBelowBar and putDistanceOk and tookPutPrev == 0;

    # Type C: 5-min outside from Inside/neutral/full-rev, with *both* five and 1-min still
    # outside within distance on the five-boundary bar (bot next-1-min after five complete).
    # Require five-distance OK so we don't mark blocked far breakouts (e.g. 10:25).
    def insideFamily = statePrev == 0 or statePrev == 1 or statePrev == 2;
    def typeCCallFire = fiveCloseBar and fiveAboveBar and oneAboveBar and callDistanceOk and
        fiveCallDistanceOk and (insideFamily or fullReversalToCall or neutralPrev == 1);
    def typeCPutFire = fiveCloseBar and fiveBelowBar and oneBelowBar and putDistanceOk and
        fivePutDistanceOk and (insideFamily or fullReversalToPut or neutralPrev == 1);

    def cleanCallFire = entriesOk and oneAboveBar and callDistanceOk and
        (sameBarCleanCall or typeAFromNeutralCall or typeCCallFire);
    def cleanPutFire = entriesOk and oneBelowBar and putDistanceOk and
        (sameBarCleanPut or typeAFromNeutralPut or typeCPutFire);
    # Pullback re-break: also respect same-side took latch.
    def pullbackCallFire = entriesOk and statePrev == 1 and oneAboveBar and callDistanceOk and tookCallPrev == 0;
    def pullbackPutFire = entriesOk and statePrev == 2 and oneBelowBar and putDistanceOk and tookPutPrev == 0;

    def pullbackMiss = (statePrev == 1 or statePrev == 2) and !pullbackCallFire and !pullbackPutFire;
    def budgetAfterOneMinute = if pullbackMiss and !fiveCloseBar then budgetPrev - 1 else budgetPrev;
    def pullbackTimedOut = pullbackMiss and !fiveCloseBar and budgetAfterOneMinute <= 0;

    # Five-minute (and session-open) own the position state — matching bot five-state.
    # Do NOT move to Outside on 1-min Type A/C fires (that poisoned later 5-min transitions).
    def nextState =
        if sessionOpen then
            if oneAboveBar then 3 else if oneBelowBar then 4 else 0
        else if pullbackTimedOut then 0
        else if fiveCloseBar then
            if fiveInsideBar then
                if statePrev == 3 then 1
                else if statePrev == 4 then 2
                else statePrev
            else if fiveAboveBar then 3
            else if fiveBelowBar then 4
            else statePrev
        else statePrev;

    def nextBudget =
        if sessionOpen then 0
        else if pullbackCallFire or pullbackPutFire or cleanCallFire or cleanPutFire then 0
        else if pullbackTimedOut then 0
        else if fiveCloseBar and fiveInsideBar and statePrev == 3 then maxTriggerCandles
        else if fiveCloseBar and fiveInsideBar and statePrev == 4 then maxTriggerCandles
        else if pullbackMiss and !fiveCloseBar then budgetAfterOneMinute
        else budgetPrev;

    def nextNeutral =
        if sessionOpen then 1
        else if fiveCloseBar and fiveInsideBar then 1
        else if fiveCloseBar and (fiveAboveBar or fiveBelowBar) then 0
        else neutralPrev;

    def nextTookCall =
        if sessionOpen then 0
        else if fiveCloseBar and fiveInsideBar then 0
        else if cleanCallFire or pullbackCallFire then 1
        else tookCallPrev;
    def nextTookPut =
        if sessionOpen then 0
        else if fiveCloseBar and fiveInsideBar then 0
        else if cleanPutFire or pullbackPutFire then 1
        else tookPutPrev;

    plot newCode = nextState + nextBudget * 10 + nextNeutral * 100 +
        nextTookCall * 1000 + nextTookPut * 10000;
}
